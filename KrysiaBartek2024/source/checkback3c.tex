Checkback 3\clubs is not forced and does not promise 5th card of the suit.
It does not deny 3/4\diams.

\subsubsection*{1\clubs -- 1\spades \\ 2\nt -- ?}
\begin{itemize}
    \item 3\clubs = checkback
    \item 3\diams = 4+\diams
    \item 3\hearts = 5\spades4\hearts
    \item 3\spades = agreeing \spades
\end{itemize}

\subsubsection*{1\clubs -- 1\spades \\ 2\nt -- 3\clubs}
\begin{itemize}
    \item 3\diams = 4+\clubs
    \item 3\hearts = 3\spades4+\clubs
    \item 3\spades = 3\spades
    \item 3\nt = no 3\spades, no 4\clubs
    \item 4\clubs/4\diams/4\hearts = 4\spades cue
\end{itemize}

\subsubsection*{1\clubs -- 1\hearts \\ 2\nt -- ?}
\begin{itemize}
    \item 3\clubs = checkback
    \item 3\diams = 4+\diams
    \item 3\hearts = 5\hearts4\spades OR agreeing \hearts
    \item 3\spades = 4\hearts4\spades
\end{itemize}

\subsubsection*{1\clubs -- 1\hearts \\ 2\nt -- 3\clubs}
\begin{itemize}
    \item 3\diams = 4+\clubs
    \item 3\hearts = 3\hearts
    \item 3\spades = 3\hearts4\clubs
    \item 3\nt = no 3\hearts, no 4\clubs
    \item 4\clubs/4\diams/4\hearts = 4\hearts cue
\end{itemize}

\subsubsection*{1\diams -- 1\spades \\ 2\nt -- ?}
\begin{itemize}
    \item 3\clubs = checkback
    \item 3\diams = 3+\diams
    \item 3\hearts = 5\spades4\hearts
    \item 3\spades = agreeing \spades
\end{itemize}

\subsubsection*{1\diams -- 1\spades \\ 2\nt -- 3\clubs}
\begin{itemize}
    \item 3\diams = 6\diams, no 3\spades
    \item 3\hearts = 3\spades6\diams
    \item 3\spades = 3\spades
    \item 3\nt = no 3\spades, no 6\diams
    \item 4\clubs/4\diams/4\hearts = 4\spades cue
\end{itemize}
